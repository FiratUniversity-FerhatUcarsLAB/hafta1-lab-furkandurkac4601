Fonksiyon KitapAra(KitapListesi, AramaTerimi):
    SonucListesi = BoşListe()
    
    for her kitap in KitapListesi:
        if AramaTerimi kitap.ad içinde veya AramaTerimi kitap.yazar içinde ise:
            SonucListesi.Ekle(kitap)
    
    return SonucListesi
